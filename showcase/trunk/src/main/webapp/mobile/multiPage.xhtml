Multiple pages can be placed in a single xhtml page, see source for the structure and navigation example for the navigation model using a custom mobile navigation handler.

Source

<!DOCTYPE html>
<html xmlns="http://www.w3.org/1999/xhtml"
	xmlns:h="http://java.sun.com/jsf/html"
	xmlns:f="http://java.sun.com/jsf/core"
	xmlns:p="http://primefaces.org/ui"
    xmlns:pm="http://primefaces.org/mobile">
    
    <f:view renderKitId="PRIMEFACES_MOBILE" />
    
    <h:head>

    </h:head>
    
    <h:body>
        
        <pm:page id="first">
            <pm:header title="Page 1"></pm:header>
            
            <pm:content>
                <p>Multiple pages can be placed in a single xhtml page.</p>
            </pm:content>
        </pm:page>

        <pm:page id="second">
            <pm:header title="Page 2"></pm:header>
            
            <pm:content>
                <p>Page 2 content.</p>
            </pm:content>
        </pm:page>

        //more pages

    </h:body>
    
</html>
