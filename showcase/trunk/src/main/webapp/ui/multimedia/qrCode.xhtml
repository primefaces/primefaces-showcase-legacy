QRCode QRCode component displays its value as a QR code. 
<h:form>
    <h:panelGrid columns="3" cellpadding="5">
        <p:outputLabel for="txt" value="Value:" />
        <p:inputText id="txt" value="\#{qrCodeView.value}" size="50"/>
        <p:commandButton value="Generate" update="qr" />
    </h:panelGrid>

    <p:outputPanel id="qr">
        <p:qrCode value="\#{qrCodeView.value}" cache="false"/>
    </p:outputPanel>
</h:form>
                
package org.primefaces.showcase.view.multimedia;

import javax.faces.bean.ManagedBean;

@ManagedBean(name="qrCodeView")
public class QRCodeView {

    private String value;

    public String getValue() {
        return value;
    }
    public void setValue(String value) {
        this.value = value;
    }
}
