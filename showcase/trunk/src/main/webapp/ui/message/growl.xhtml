Message - Growl Growl brings the Mac's growl widget to JSF with the ability of displaying FacesMessages. Growl simply replaces h:messages component. 
<h:form>  
<p:growl id="growl" showDetail="true" sticky="true" />  
      
<p:panel header="Growl">  
<h:panelGrid columns="2">  
<h:outputText value="Your Message: *" />   
<p:inputText value="\#{growlView.message}" required="true" label="Message"/>  
</h:panelGrid>  
  
<p:commandButton value="Save" actionListener="\#{growlView.save}" update="growl"/>  
</p:panel>  
</h:form> 
                
package org.primefaces.showcase.view.message;

import javax.faces.application.FacesMessage;
import javax.faces.bean.ManagedBean;
import javax.faces.context.FacesContext;
import javax.faces.event.ActionEvent;

@ManagedBean
public class GrowlView {
    
    private String message;

    public String getMessage() {
        return message;
    }

    public void setMessage(String message) {
        this.message = message;
    }
    
    public void saveMessage(ActionEvent actionEvent) {
        FacesContext context = FacesContext.getCurrentInstance();
        
        context.addMessage(null, new FacesMessage("Successful",  "Your message : "+message) );
        context.addMessage(null, new FacesMessage("Second Message", "Additional Message Here..."));
    }
}
